import pandas as pd
import numpy as np
import yfinance as yf
import matplotlib.pyplot as plt
from statsmodels.tsa.stattools import adfuller

# Function to download stock data
def download_stock_data(ticker, start_date, end_date):
    stock_data = yf.download(ticker, start=start_date, end=end_date)
    return stock_data

# Function to calculate moving averages
def calculate_moving_averages(stock_data, short_window, long_window):
    stock_data['MA9'] = stock_data['Close'].rolling(window=short_window).mean()
    stock_data['MA30'] = stock_data['Close'].rolling(window=long_window).mean()
    return stock_data

# Function to perform the Augmented Dickey-Fuller test
def adf_test(series):
    result = adfuller(series, autolag='AIC')
    return result

# Function to plot stock data with moving averages
def plot_stock_data_with_moving_averages(stock_data):
    plt.figure(figsize=(14, 7))
    plt.plot(stock_data['Close'], label='Close Price')
    plt.plot(stock_data['MA9'], label='9-Day MA')
    plt.plot(stock_data['MA30'], label='30-Day MA')
    plt.title('Home Depot (HD) Stock Price with Moving Averages')
    plt.xlabel('Date')
    plt.ylabel('Stock Price')
    plt.legend()
    plt.show()

# Main function to run the analysis
def main():
    ticker = 'HD'
    start_date = '2019-11-29'
    end_date = '2025-02-25'

    # Download stock data
    stock_data = download_stock_data(ticker, start_date, end_date)

    # Calculate moving averages
    stock_data = calculate_moving_averages(stock_data, short_window=9, long_window=30)

    # Perform the Augmented Dickey-Fuller test
    adf_result = adf_test(stock_data['Close'].dropna())
    print('ADF Statistic:', adf_result[0])
    print('p-value:', adf_result[1])
    for key, value in adf_result[4].items():
        print(f'Critical Value ({key}): {value}')

    # Plot stock data with moving averages
    plot_stock_data_with_moving_averages(stock_data)

if __name__ == '__main__':
    main()

import pandas as pd
import numpy as np
import yfinance as yf
import matplotlib.pyplot as plt
from statsmodels.tsa.stattools import adfuller

# Function to download stock data
def download_stock_data(ticker, start_date, end_date):
    stock_data = yf.download(ticker, start=start_date, end=end_date)
    return stock_data

# Function to calculate moving averages
def calculate_moving_averages(stock_data, short_window, long_window):
    stock_data['MA9'] = stock_data['Close'].rolling(window=short_window).mean()
    stock_data['MA30'] = stock_data['Close'].rolling(window=long_window).mean()
    return stock_data

# Function to perform the Augmented Dickey-Fuller test
def adf_test(series):
    result = adfuller(series, autolag='AIC')
    return result

# Function to generate buy alerts
def generate_buy_alerts(stock_data):
    stock_data['Buy_Signal'] = np.where(stock_data['MA9'] > stock_data['MA30'], 1, 0)
    stock_data['Buy_Signal'] = stock_data['Buy_Signal'].diff()
    return stock_data

# Function to plot stock data with moving averages and buy alerts
def plot_stock_data_with_moving_averages_and_buy_alerts(stock_data):
    plt.figure(figsize=(14, 7))
    plt.plot(stock_data['Close'], label='Close Price')
    plt.plot(stock_data['MA9'], label='9-Day MA')
    plt.plot(stock_data['MA30'], label='30-Day MA')
    plt.plot(stock_data[stock_data['Buy_Signal'] == 1].index, stock_data['MA9'][stock_data['Buy_Signal'] == 1], '^', markersize=10, color='g', label='Buy Signal')
    plt.title('Apple Inc. (AAPL) Stock Price with Moving Averages and Buy Alerts')
    plt.xlabel('Date')
    plt.ylabel('Stock Price')
    plt.legend()
    plt.show()

    ticker = 'AAPL'

    # Download stock data
